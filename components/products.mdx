# Products Carousel

The Products Carousel displays water bottle presentations in a responsive layout - a grid on desktop and a touch-friendly carousel on mobile.

## Location

- **Component**: `components/productsCarousel.tsx`
- **Data**: `components/products.tsx`

## Features

- Responsive design (grid on desktop, carousel on mobile)
- Touch-friendly swiping with Embla Carousel
- Hover zoom effects on images
- Price formatting utility
- Badge integration for delivery info
- Dot indicators for carousel navigation

## Component Structure

```tsx
import { BadgeCheckIcon } from "lucide-react";
import { Products } from "./products";
import { formatPrice } from "./utils/formatters";
import { Badge } from "@/components/ui/badge";
import useEmblaCarousel from "embla-carousel-react";
import { useCallback, useEffect, useState } from "react";
import clsx from "clsx";

function PresentationsCarousel() {
  const [emblaRef, emblaApi] = useEmblaCarousel({ loop: false });
  const [selectedIndex, setSelectedIndex] = useState(0);
  const [scrollSnaps, setScrollSnaps] = useState<number[]>([]);

  // Carousel controls
  const scrollTo = useCallback(
    (index: number) => {
      emblaApi && emblaApi.scrollTo(index);
    },
    [emblaApi]
  );

  const onSelect = useCallback(() => {
    if (!emblaApi) return;
    setSelectedIndex(emblaApi.selectedScrollSnap());
  }, [emblaApi]);

  useEffect(() => {
    if (!emblaApi) return;
    setScrollSnaps(emblaApi.scrollSnapList());
    emblaApi.on("select", onSelect);
    onSelect();
    return () => {
      emblaApi.off("select", onSelect);
    };
  }, [emblaApi, onSelect]);

  return (
    <>
      {/* Desktop Grid */}
      {/* Mobile Carousel */}
    </>
  );
}
```

## Mobile Carousel Layout

```tsx
<div className="block md:hidden relative">
  <div className="overflow-hidden py-8" ref={emblaRef}>
    <div className="flex">
      {Products.map((product, i) => (
        <div key={i} className="flex-[0_0_70%] pl-5 pr-5">
          <div className="group h-90 flex flex-col text-start rounded-[40px] bg-white overflow-hidden shadow-[4px_4px_8px_#d1d1d1,-4px_-4px_8px_#ffffff] w-full will-change-transform">
            {/* Image */}
            <div className="overflow-hidden rounded-t-[40px]">
              <div className="transition-transform duration-500 ease-out">
                {product.image}
              </div>
            </div>

            <div className="relative -top-42 left-24">
              <Badge variant="secondary" className="bg-blue-500 text-white text-xs">
                <BadgeCheckIcon className="w-3 h-3 mr-1" />
                {product.amount}
              </Badge>
            </div>

            {/* Content */}
            <div className="h-45 px-6 pb-6 flex flex-col justify-between">
              <div className="flex flex-col items-start gap-3 mb-4">
                <h3 className="text-base font-semibold text-[#1F01B9]">
                  {product.title}
                </h3>
                <p className="text-gray-700 text-xs leading-relaxed">
                  {product.description}
                </p>
              </div>
              <div className="flex flex-col">
                <p className="text-lg font-extrabold text-blue-600">
                  {formatPrice(product.price)}
                </p>
              </div>
            </div>
          </div>
        </div>
      ))}
    </div>
  </div>

  {/* Dot indicators */}
  <div className="flex justify-center gap-2 -mt-2">
    {Products.map((_, i) => (
      <button
        key={i}
        onClick={() => scrollTo(i)}
        className={clsx(
          "w-2 h-2 rounded-full transition-all",
          i === selectedIndex ? "bg-[#1F01B9] w-4" : "bg-gray-300"
        )}
        aria-label={`Ir a producto ${i + 1}`}
      />
    ))}
  </div>
</div>
```

## Price Formatting

Location: `components/utils/formatters.ts`

```tsx
export const formatPrice = (price: number): string => {
  return `S/ ${price.toFixed(2)}`;
};
```

Usage:

```tsx
import { formatPrice } from "./utils/formatters";

<p className="text-xl font-bold text-blue-600">
  {formatPrice(product.price)}
</p>
```
